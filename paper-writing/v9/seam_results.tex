% Every number produced by the GPU training program (T0 prompt-era
% baselines, T1 SFT, T2 GRPO, and the judge-panel calibration gate) is
% declared here as a \newcommand that defaults to a visible \pending
% marker.  The seam section's results tables (\S\ref{sec:seam}) and the
% E5 translation-fidelity cells (\S\ref{sec:bench}) consume ONLY these
% macros -- nothing is hardcoded.  Post-GPU, fill in a number by editing
% exactly one \renewcommand line below; the paper re-typesets with no
% structural change.  TEMPLATE_HOWTO.md maps each macro to the harness
% output field (experiments/seam_bench/eval metrics) that fills it.
%
% Discipline: DO NOT edit the tables in main.tex to drop a number in.
% Edit only this file.  A macro still showing \pending in the compiled
% PDF is an honest, visible "not yet measured" cell -- the same
% pending-cell discipline the rest of the paper follows.
% =====================================================================

% The visible marker for any not-yet-measured cell.  One definition,
% one place to restyle every pending cell at once.
\newcommand{\pending}{{\small\textsc{pending}}}

% Convenience: use \seamfill{<macro>}{<value>} in a "fill sheet" if you
% prefer; equivalently \renewcommand the target directly.  Provided so
% the HOWTO can show a single canonical fill idiom.
\newcommand{\seamfill}[2]{\renewcommand{#1}{#2}}
