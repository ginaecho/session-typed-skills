global protocol PrReviewMerge(role Author, role CodeReviewer, role SecurityReviewer, role Merger) {
    ReadyForReview(String) from Author to CodeReviewer;
    ReadyForReview(String) from Author to SecurityReviewer;

    rec ROUND {
        choice at CodeReviewer {
            ReviewComments(String) from CodeReviewer to Author;
            ReviewComments(String) from CodeReviewer to SecurityReviewer;
            ReviewComments(String) from CodeReviewer to Merger;
            Revision(String) from Author to CodeReviewer;
            Revision(String) from Author to SecurityReviewer;
            continue ROUND;
        } or {
            QualityClean(String) from CodeReviewer to SecurityReviewer;
            QualityClean(String) from CodeReviewer to Author;
            QualityClean(String) from CodeReviewer to Merger;
            choice at SecurityReviewer {
                SecurityFindings(String) from SecurityReviewer to Author;
                SecurityFindings(String) from SecurityReviewer to CodeReviewer;
                SecurityFindings(String) from SecurityReviewer to Merger;
                Revision(String) from Author to CodeReviewer;
                Revision(String) from Author to SecurityReviewer;
                continue ROUND;
            } or {
                SecurityApproved(String) from SecurityReviewer to Merger;
                SecurityApproved(String) from SecurityReviewer to CodeReviewer;
                SecurityApproved(String) from SecurityReviewer to Author;
                QualityApproved(String) from CodeReviewer to Merger;
            }
        }
    }

    MergeDone(String) from Merger to Author;
}

Author

agents/address-comments.agent.md
"Your job is to address comments on your pull request. Fix each comment, add tests, commit. You are done when your change has been merged."
⚠ Never told to whom revisions go, or who must approve what.

CodeReviewer

instructions/code-review-generic.instructions.md
"Review line-by-line with severity triage; 🔴 CRITICAL findings block the merge."
⚠ "Block merge" — but nothing wires its verdict to the Merger.

SecurityReviewer

agents/se-security-reviewer.agent.md
"Review against the OWASP Top 10; report Ready for Production: Yes/No."
⚠ Writes a report file — no recipient is ever named.

Merger

agents/principal-software-engineer.agent.md
"Make the ship call; bias to delivery, keep the team unblocked."
⚠ Nothing says the two reviews must come first.
